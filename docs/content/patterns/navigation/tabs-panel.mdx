# Tabs Panel

A tabbed interface for organizing content into separate views. Built using `Box`, `Row`, `Text`, and context for state management.

<Example>
  <TabsPanel defaultTab="overview">
    <TabsList>
      <Tab id="overview">Overview</Tab>
      <Tab id="analytics">Analytics</Tab>
      <Tab id="settings">Settings</Tab>
    </TabsList>
    <TabContent id="overview">
      <Text>Overview content goes here.</Text>
    </TabContent>
    <TabContent id="analytics">
      <Text>Analytics content goes here.</Text>
    </TabContent>
    <TabContent id="settings">
      <Text>Settings content goes here.</Text>
    </TabContent>
  </TabsPanel>

## Source

```jsx
'use client';

import { createContext, useContext, useState } from 'react';
import { Box, Row, Text } from '@umami/react-zen';

const TabsPanelContext = createContext({
  activeTab: '',
  setActiveTab: () => {},
});

export function useTabsPanel() {
  return useContext(TabsPanelContext);
}

export function TabsPanel({ defaultTab, children, ...props }) {
  const [activeTab, setActiveTab] = useState(defaultTab);

  return (
    <TabsPanelContext.Provider value={{ activeTab, setActiveTab }}>
      <Box {...props}>{children}</Box>
    </TabsPanelContext.Provider>
  );
}

export function TabsList({ children }) {
  return (
    <Row gap="0" borderColor="muted" border="bottom">
      {children}
    </Row>
  );
}

export function Tab({ id, children }) {
  const { activeTab, setActiveTab } = useTabsPanel();
  const isActive = activeTab === id;

  return (
    <Box
      paddingX="4"
      paddingY="3"
      className="cursor-pointer"
      onClick={() => setActiveTab(id)}
      style={{
        borderBottom: isActive ? '2px solid var(--color-primary)' : '2px solid transparent',
        marginBottom: '-1px',
      }}
    >
      <Text size="sm" weight={isActive ? 'medium' : undefined} color={isActive ? undefined : 'muted'}>
        {children}
      </Text>
    </Box>
  );
}

export function TabContent({ id, children, ...props }) {
  const { activeTab } = useTabsPanel();

  if (activeTab !== id) return null;

  return (
    <Box padding="4" {...props}>
      {children}
    </Box>
  );
}
```
