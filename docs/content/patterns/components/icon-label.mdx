# Icon Label

A simple pattern for pairing an icon with text. This is one of the most common UI patterns and is easily composed using the `Row`, `Icon`, and `Text` components.

<Example>
  <Row alignItems="center" gap="2">
    <Icon><Lucide.Settings /></Icon>
    <Text>Settings</Text>
  </Row>
</Example>

## Basic implementation

Copy this code into your project:

```tsx
'use client';

import { Icon, Row, Text } from '@umami/react-zen';
import { Settings } from 'lucide-react';

export function IconLabel() {
  return (
    <Row alignItems="center" gap="2">
      <Icon>
        <Settings />
      </Icon>
      <Text>Settings</Text>
    </Row>
  );
}
```

## Reusable component

For a reusable version with props:

```tsx
'use client';

import type { ReactNode } from 'react';
import { Icon, Row, Text, type RowProps, type IconProps, type TextProps } from '@umami/react-zen';

interface IconLabelProps extends RowProps {
  icon: ReactNode;
  label?: ReactNode;
  iconProps?: Partial<IconProps>;
  labelProps?: Partial<TextProps>;
}

export function IconLabel({ icon, label, iconProps, labelProps, children, ...props }: IconLabelProps) {
  return (
    <Row alignItems="center" gap="2" {...props}>
      <Icon {...iconProps}>{icon}</Icon>
      {label && <Text {...labelProps}>{label}</Text>}
      {children}
    </Row>
  );
}
```

Usage:

```tsx
import { IconLabel } from './IconLabel';
import { Settings, User, Mail } from 'lucide-react';

// Basic usage
<IconLabel icon={<Settings />} label="Settings" />

// With custom styling
<IconLabel
  icon={<User />}
  label="Profile"
  iconProps={{ color: 'blue' }}
  labelProps={{ weight: 'bold' }}
/>

// With children instead of label
<IconLabel icon={<Mail />}>
  <Text>Messages</Text>
  <Text color="muted" size="sm">(3 unread)</Text>
</IconLabel>
```
