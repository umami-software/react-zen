# Nav Menu

A vertical navigation menu with collapsible groups, commonly used in sidebars and dashboards. Built using `Column`, `Row`, `Text`, and `Icon` components.

## Full implementation

Copy this code into your project for a fully functional nav menu with collapsible groups:

```tsx
'use client';

import { createContext, useContext, useState, type ReactNode } from 'react';
import { Column, Row, Text, Icon, type ColumnProps, type RowProps } from '@umami/react-zen';
import { ChevronRight } from 'lucide-react';

// Context for sharing state
interface NavMenuContextValue {
  onItemClick?: () => void;
}

const NavMenuContext = createContext<NavMenuContextValue>({});

// Main NavMenu container
interface NavMenuProps extends ColumnProps {
  onItemClick?: () => void;
}

export function NavMenu({ onItemClick, children, ...props }: NavMenuProps) {
  return (
    <NavMenuContext.Provider value={{ onItemClick }}>
      <Column gap="1" {...props}>
        {children}
      </Column>
    </NavMenuContext.Provider>
  );
}

// Collapsible group
interface NavMenuGroupProps extends ColumnProps {
  title: string;
  defaultOpen?: boolean;
}

export function NavMenuGroup({ title, defaultOpen = true, children, ...props }: NavMenuGroupProps) {
  const [isOpen, setIsOpen] = useState(defaultOpen);

  return (
    <Column {...props}>
      <Row
        paddingY="2"
        paddingX="3"
        alignItems="center"
        justifyContent="space-between"
        className="cursor-pointer"
        onClick={() => setIsOpen(!isOpen)}
      >
        <Text size="xs" weight="semibold" color="muted" transform="uppercase">
          {title}
        </Text>
        <Icon size="sm" color="muted" rotate={isOpen ? 90 : 0}>
          <ChevronRight />
        </Icon>
      </Row>
      {isOpen && <Column>{children}</Column>}
    </Column>
  );
}

// Individual menu item
interface NavMenuItemProps extends RowProps {
  icon?: ReactNode;
  isSelected?: boolean;
}

export function NavMenuItem({ icon, isSelected, children, ...props }: NavMenuItemProps) {
  const { onItemClick } = useContext(NavMenuContext);

  return (
    <Row
      paddingY="2"
      paddingX="3"
      borderRadius="md"
      alignItems="center"
      gap="2"
      onClick={onItemClick}
      backgroundColor={isSelected ? 'interactive' : undefined}
      className={`cursor-pointer ${!isSelected ? 'hover:bg-interactive' : ''}`}
      {...props}
    >
      {icon && (
        <Icon size="sm" color="muted">
          {icon}
        </Icon>
      )}
      <Text color={isSelected ? undefined : 'muted'} weight={isSelected ? 'medium' : undefined}>
        {children}
      </Text>
    </Row>
  );
}
```

## Usage example

```tsx
'use client';

import { useState } from 'react';
import { Box } from '@umami/react-zen';
import { Home, BarChart, Users, Settings, Shield } from 'lucide-react';
import { NavMenu, NavMenuGroup, NavMenuItem } from './NavMenu';

export function Sidebar() {
  const [selected, setSelected] = useState('home');

  return (
    <Box width="64" border borderRadius="lg" padding="2">
      <NavMenu onItemClick={() => console.log('Item clicked')}>
        <NavMenuGroup title="Dashboard">
          <NavMenuItem
            icon={<Home />}
            isSelected={selected === 'home'}
            onClick={() => setSelected('home')}
          >
            Home
          </NavMenuItem>
          <NavMenuItem
            icon={<BarChart />}
            isSelected={selected === 'analytics'}
            onClick={() => setSelected('analytics')}
          >
            Analytics
          </NavMenuItem>
          <NavMenuItem
            icon={<Users />}
            isSelected={selected === 'users'}
            onClick={() => setSelected('users')}
          >
            Users
          </NavMenuItem>
        </NavMenuGroup>

        <NavMenuGroup title="Settings">
          <NavMenuItem
            icon={<Settings />}
            isSelected={selected === 'general'}
            onClick={() => setSelected('general')}
          >
            General
          </NavMenuItem>
          <NavMenuItem
            icon={<Shield />}
            isSelected={selected === 'security'}
            onClick={() => setSelected('security')}
          >
            Security
          </NavMenuItem>
        </NavMenuGroup>
      </NavMenu>
    </Box>
  );
}
```

## Variations

### Without icons

```tsx
<NavMenu>
  <NavMenuGroup title="Pages">
    <NavMenuItem isSelected>Dashboard</NavMenuItem>
    <NavMenuItem>Reports</NavMenuItem>
    <NavMenuItem>Settings</NavMenuItem>
  </NavMenuGroup>
</NavMenu>
```

### Non-collapsible groups

Simply don't use `NavMenuGroup` and use a static header:

```tsx
<NavMenu>
  <Row paddingY="2" paddingX="3">
    <Text size="xs" weight="semibold" color="muted" transform="uppercase">
      Navigation
    </Text>
  </Row>
  <NavMenuItem icon={<Home />} isSelected>Home</NavMenuItem>
  <NavMenuItem icon={<Settings />}>Settings</NavMenuItem>
</NavMenu>
```

### With badges

```tsx
<NavMenuItem icon={<Mail />}>
  <Row alignItems="center" justifyContent="space-between" width="100%">
    <Text>Messages</Text>
    <Box
      backgroundColor="red-500"
      paddingX="2"
      paddingY="0.5"
      borderRadius="full"
    >
      <Text size="xs" color="white">3</Text>
    </Box>
  </Row>
</NavMenuItem>
```
