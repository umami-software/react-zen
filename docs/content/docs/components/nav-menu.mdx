# Nav Menu

A collection of navigation links.

<Example>
  <Box width="200px">
    <NavMenu>
      <NavMenuItem isSelected>One</NavMenuItem>
      <NavMenuItem>Two</NavMenuItem>
      <NavMenuItem>Three</NavMenuItem>
      <NavMenuItem>Four</NavMenuItem>
      <NavMenuItem>Five</NavMenuItem>
    </NavMenu>
  </Box>
</Example>

## Groups

<Example>
  <Box width="200px">
    <NavMenu>
      <NavMenuGroup title="Group 1">
        <NavMenuItem>One</NavMenuItem>
        <NavMenuItem>Two</NavMenuItem>
        <NavMenuItem>Three</NavMenuItem>
        <NavMenuItem>Four</NavMenuItem>
        <NavMenuItem>Five</NavMenuItem>
      </NavMenuGroup>
      <NavMenuGroup title="Group 2" allowMinimize={false}>
        <NavMenuItem>One</NavMenuItem>
        <NavMenuItem>Two</NavMenuItem>
        <NavMenuItem>Three</NavMenuItem>
        <NavMenuItem>Four</NavMenuItem>
        <NavMenuItem>Five</NavMenuItem>
      </NavMenuGroup>
    </NavMenu>
  </Box>
</Example>
